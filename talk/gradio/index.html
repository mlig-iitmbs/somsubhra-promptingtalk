    Gradio Sample NB

    • Gradio is basically an open-source library that allows you to create a shareable interactive web interface from your Python code without the need of HTML/CSS/JS. You may go through this Medium blog - nicely explained!.

    • Here we are creating a simple interactive 5-Round Palindrome Quiz game.

    The application functions as follows:

    1. It presents a series of five random words and asks the user to classify each word as a Palindrome or Not a Palindrome using dedicated “Yes” or “No” buttons.
    2. Keeps track of the current round number and the user’s running score.
    3. For each of the five rounds, it logs the following data- the Word presented, user’s answer (Yes/No) and the Correct Answer (Yes/No).
    4. Whether the user was Correct (True/False).
    5. After all five rounds are complete, a “Save Results to CSV” button appears. Clicking this button compiles all the logged data into a pandas DataFrame and saves it as a file named palindrome_quiz_results.csv in the Colab nb directory.

import gradio as gr
import pandas as pd
import random
import os

WORD_LIST = ["level", "hello", "rotor", "world", "madam", "python", "coding", "refer", "data"]

def is_palindrome(word):
    return word.lower() == word.lower()[::-1]

def start_quiz():
    initial_state = ["", 0, []]

    next_word = random.choice(WORD_LIST)

    initial_state[0] = next_word
    initial_state[1] = 1

    return (
        initial_state,
        f"Question 1/5: Is '{next_word.upper()}' a palindrome?",
        gr.update(visible=True),
        gr.update(visible=False)
    )

def answer_question(answer, state):
    current_word, current_round, results = state

    correct_answer = "Yes" if is_palindrome(current_word) else "No"
    is_correct = (answer == correct_answer)

    results.append({
        'Round': current_round,
        'Word': current_word,
        'User_Answer': answer,
        'Correct_Answer': correct_answer,
        'Correct': is_correct
    })

    next_round = current_round + 1

    if next_round > 5:
        final_state = ["", next_round, results]
        score = sum(r['Correct'] for r in results)

        return (
            final_state,
            f"Quiz Complete! Your Score: {score}/5",
            gr.update(visible=False),
            gr.update(visible=True)
        )
    else:
        next_word = random.choice(WORD_LIST)
        new_state = [next_word, next_round, results]

        return (
            new_state,
            f"Question {next_round}/5: Is '{next_word.upper()}' a palindrome?",
            gr.update(visible=True),
            gr.update(visible=False)
        )

def save_results(state):
    _, _, results = state
    if not results:
        return "No results to save!"

    df = pd.DataFrame(results)
    file_name = "palindrome_quiz_results.csv"

    df.to_csv(file_name, index=False)

    return f"Results saved successfully to **{os.path.abspath(file_name)}** in your Colab environment."


with gr.Blocks(title="Palindrome Quiz") as demo:
    gr.Markdown("## 🔡 Palindrome Quiz 5-Round Challenge")

    quiz_state = gr.State(value=["", 0, []])

    question_display = gr.Markdown("Click 'Start Quiz' to begin.")
    status_message = gr.Textbox(label="Status", value="", interactive=False)

    with gr.Row():
        start_btn = gr.Button("Start Quiz 🚀", variant="primary")

    with gr.Column(visible=False) as quiz_column:
        with gr.Row():
            yes_btn = gr.Button("Yes (Palindrome)")
            no_btn = gr.Button("No (Not Palindrome)")

    save_btn = gr.Button("💾 Save Results to CSV", visible=False, variant="secondary")
    save_output = gr.Textbox(label="Save Status", interactive=False)


    start_btn.click(
        fn=start_quiz,
        inputs=None,
        outputs=[quiz_state, question_display, quiz_column, save_btn],
        show_progress=False
    )

    yes_btn.click(
        fn=answer_question,
        inputs=[gr.State(value="Yes"), quiz_state],
        outputs=[quiz_state, question_display, quiz_column, save_btn],
        show_progress=False
    )

    no_btn.click(
        fn=answer_question,
        inputs=[gr.State(value="No"), quiz_state],
        outputs=[quiz_state, question_display, quiz_column, save_btn],
        show_progress=False
    )

    save_btn.click(
        fn=save_results,
        inputs=quiz_state,
        outputs=save_output
    )

demo.launch(inbrowser=True)
